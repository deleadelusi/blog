# Load the ggplot2 package
library(ggplot2)
library(tidyverse)

## Download COVID data from OWID GitHub
owidall = read.csv("https://github.com/owid/covid-19-data/blob/master/public/data/owid-covid-data.csv?raw=true")
# Deselect cases/rows with OWID
owidall1 = owidall[!grepl("^OWID", owidall$iso_code), ]
# Subset by continent: Europe
owideu = subset(owidall1, continent=="Europe")
#change the class to date -> useful to filter later
owideu$date <- as.Date(owideu$date)

library(dplyr)

# dates so I can have thresholds to put labels on
specific_dates <- c("2020-12-20", "2020-04-05", "2020-03-29", "2021-11-14", "2023-01-01", "2023-08-31")

# make a df filtering the specific dates
label_data <- owideu %>%
  filter(date %in% specific_dates) %>%
  group_by(date) %>%
  arrange(desc(new_deaths)) %>%
  slice(1) %>%
  ungroup()

# plot the data changing all the settings
ggplot(data = owideu, aes(x = date, y = new_deaths)) +
  geom_point(size = 1, color = "#f00276") +
  geom_text(data = label_data, aes(label = location), vjust = -0.5, family="serif") +  
  theme(
    text = element_text(family = "serif", size = 12),
    axis.text.x = element_text(angle = 90, hjust = 1, family = "serif"),
    axis.text.y = element_text(angle = 90, hjust = 1, family = "serif"),
    panel.background = element_rect(fill = "white")
  ) +
  ylab("COVID Deaths in Europe (Daily)") +
  xlab("Date") +
  scale_x_date(
    date_breaks = "2 months",
    date_labels = "%Y %m",
    minor_breaks = "1 month"
  )
